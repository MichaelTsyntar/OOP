create database Airlines
go
use Airlines
go
CREATE TABLE Passenger (
    Id int NOT NULL IDENTITY  PRIMARY KEY,
    Name varchar(20)NOT NULL,
    LastName varchar(20)NOT NULL
)
GO

CREATE TABLE Flight_schedule ( 
      Id int Not null IDENTITY PRIMARY KEY,         
      Departure_date varchar(30) NOT NULL,
      Departure_time varchar(30) NOT NULL   
 )
GO

CREATE TABLE Flight( 
      Id int NOT NULL  IDENTITY  PRIMARY KEY, 
      Flight_shedule_Id int Not Null ,
      Passenger_Id int Not Null ,
      Froms varchar(30)NOT NULL ,
      Wheres varchar(30) NOT NULL,
      Airlines varchar(30) NOT NULL,    
      FOREIGN KEY (Passenger_Id)  REFERENCES Passenger (Id) ,
      FOREIGN KEY ( Flight_shedule_Id)  REFERENCES Flight_schedule (Id) 
) 

GO
